@EndUserText.label: 'ZSTR_PO_CLOSE - Data structure for PO closure processing'
define structure ZSTR_PO_CLOSE {
  EBELN      : abap.char(10);    -- Purchase Order Number
  EBELP      : abap.numc(5);     -- Item Number
  MATNR      : abap.char(40);    -- Material Number
  LIFNR      : abap.char(10);    -- Vendor
  MENGE      : abap.quan(13,3);  -- Order Quantity
  WEMNG      : abap.quan(13,3);  -- Goods Received Quantity
  OPEN_QTY   : abap.quan(13,3);  -- Open Quantity
  AEDAT      : abap.dats;        -- Last Change Date
  FLAG_CLOSE : abap.char(1);     -- User Selection for Closure
  MESSAGE    : abap.char(100);   -- System Message after closure
}
